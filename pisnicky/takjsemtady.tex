\hyt{takjsemtady}
\song{Tak jsem tady}

\vers{1}{
\chord{D}Ani jsem \chord{Hm}nepřijel \chord{Em}na slupce \chord{A\7}od baná\chord{D}nu,\\
\chord{D}ani se nedá říct, \chord{E\7}že to šlo podle plá\chord{A\7}nu.\\
\chord{F\kk}Trochu mě zdržela \chord{Hm}důvěra v \chord{Em}jízdní řá\chord{A\7}dy,\\
ale \chord{D}ten, kdo ví, \chord{Hm}kam jede, \chord{Em}vždycky si \chord{A\7}už pora\chord{D}dí.
}

\vers{2}{
Já nemám na taxi, a tak si jezdím stopem,\\
jedni se courají a druzí to berou hopem.\\
Člověk to musí brát podle těch, co ho berou,\\
každý má víc tváří, jenom si vybrat, kterou.
}

\vers{3}{
Už jsem si zvykl, že vždycky mě někdo vezme,\\
dívám se z okna a snažím se hádat, kde jsme.\\
Krajina většinou zdá se mi povědomá,\\
asi to bude tím, že jsem tu někde doma.
}

\vers{4}{
Všude jsem doma a všude jsem trochu cizí,\\
sleduj v zrcátku krajinu, než mi zmizí.\\
A říkám si, kde asi, kde domov můj leží?\\
\rep{Ale vždyť na tom nakonec zas tolik nezáleží.}
}
