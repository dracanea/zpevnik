\hyt{pocestny}
\song{Pocestný}
\vers{1}{
\chord{G}Je to chůze \chord{G\7}po tom světě, \chord{C}kam se noha \chord{G}šine,\\
\rep{\chord{D\7}sotva p\chord{D\kk\dm}řejdeš\mm \chord{Em}jedny ho\chord{C}ry, \chord{D\add{7}}hned se najdou \chord{G}jiné.}
}

\vers{2}{
Je to chůze po tom světě, že by člověk utek,\\
\rep{ještě nezažil jsi jeden, máš tu druhý smutek.}
}

\vers{3}{
Což je pánům v krytém voze, sedí pěkně v suše,\\
\rep{ale chudý, ten za nimi v dešti, blátě kluše.}
}

\vers{4}{
Ej, co já dbám na té cestě na psoty a sloty,\\
\rep{jen když já mám zdravé nohy, k tomu dobré boty.}
}

\vers{5}{
Však na pány v krytém voze taky někdy trhne,\\
\rep{jednou se jim kolo zláme, jindy vůz se zvrhne.}
}

\vers{6}{
A krom toho, až své pouti přejedem a přejdem,\\
\rep{v jedné hospodě na nocleh pán nepán se sejdem}
}
