\vers{1}{
\chord{G}Černý muž pod bičem otrokáře \chord{G\7}žil\\
\chord{C}černý muž pod bičem \chord{G}otrokáře žil,\\

kapi\chord{Am\7}tán John \chord{D\7}Brown to \chord{G}zřel.\chord{(D\7)}
}

\refrain{
Glory, glory aleluja,\\
glory, glory aleluja,\\
glory, glory aleluja,\\
kapitán John Brown to zřel.
}

\vers{2}{
\rep{Sebral z Virginie\\
	věrných přátel šik}\\
prapor svobody tam zdvih.
}\refsm{}

\vers{3}{
\rep{V čele věrných město\\
Herpers-Ferry jal,}\\
právo, vítězství a čest.
}\refsm{}

\vers{4}{
\rep{Hrstka statečných však\\
	udolána jest,}\\
kapitán John Brown je jat.
}\refsm{}

\vers{5}{
\rep{Zvony Charles Townu\\
z dáli temně zní,}\\
poslední to Johnův den.
}\refsm{}

\vers{6}{
\rep{John Brown mrtev jest\\
a jeho tělo tlí,}\\
jeho duch však kráčí dál.
}\refrain{}{}

\centerline{\emph{verze Waldemara Matušky}}

\vers{1}{
Dávno svět už jméno Johna Browna zná,\\
zní jak polnice a trubka vítězná.\\
V záři táborových ohňů zas a zas\\
tahle píseň zdvíhá hlas.
}

\refrain{
Glory, glory aleluja,\\
glory, glory aleluja,\\
glory, glory aleluja,\\
jménu kapitán John Brown.
}

\vers{2}{
\rep{V bídě okovů žil celý černý Jih,}\\
až k boji za svobodu houfy statečných\\
zvedl kapitán John Brown.
}\refsm{}

\vers{3}{
\rep{Vlajka nad hlavou jim\\
	sotva začla vlát,}\\
zákeřnou přesilou je poražen a jat\\
v boji kapitán John Brown.
}\refsm{}

\vers{4}{
\rep{Z rána v Charles Townu\\
	umíráček zní,}\\
před popravčí četou ve své chvíli poslední\\
stojí kapitán John Brown.
}\refsm{}

\vers{5}{
John Brown je mrtev, ale živ je ideál,\\
ten, za nímž doposledka jako stála stál.\\
John Brown je mrtev,\\
ale pravda, které přál,\\
jako slunce září dál.
}\refsm{}
