\hyt{kozel}
\song{Kozel}

\vers{1}{
\chord{G}Byl jeden pán, ten kozla \chord{C}měl, velice \chord{D}si s ním rozu\chord{G}měl.\\
Měl ho moc \chord{G}rád, opravdu \chord{C}moc, hladil mu \chord{D}fous na dobrou \chord{G}noc.
}

\vers{2}{
Jednoho dne se kozel splet', rudé tričko pánovi sněd'.\\
Když to pán zřel, zařval jéjé, svázal kozla na koleje.
}

\vers{3}{
Zahoukal vlak, kozel se lek', \uv{To je má smrt!}, mečel, mek, mek.\\
Jak tak mečel, vykašlal pak rudé tričko, čímž stopnul vlak.
}

\vers{4}{
Na, na, na, na\dots
}

\vers{5}{
Byl jeden pán, ten šnečka měl, velice si s ním rozuměl.\\
Měl ho moc rád, opravdu moc, hladil mu ulitu na dobrou noc.
}

\vers{6}{
Jednoho dne se šneček splet', zelené tričko pánovi sněd'.\\
Když to pán zřel, zařval jéjé, svázal šněčka na koleje.
}

\vers{7}{
Zahoukal vlak, šneček se lek', \uv{To je má smrt!}, šnekal, šnek, šnek.\\
Jak tak šnekal, vykašlal pak zelené tričko, čímž zrychlil vlak.
}

\vers{8}{
Křup, křup, křup, křup\dots
}
