\hyt{letosmonikou}
\song{Léto s Monikou}
\hyl{inthesummertime}{In the Summertime}
\ns

\vers{1}{
Jeden \chord{A}letní den, kdy jsem nemohl spát\\
jako krásný sen, který měl se mi zdát,\\
přišla \chord{D}ke mně sem dívka bezvadná a zůstala tu \chord{A}stát.\\
\uv{Já jsem \chord{E}Monika,} řekla, \chord{D}a já už ji měl \chord{A}rád.
}

\vers{2}{
Léto s Monikou bylo stoprocentní,\\
láskou velikou totiž hořel jsem k ní,\\
celý prázdniny měly nádhernej a trochu rychlej spád.\\
\uv{Pojď mě políbit,} řekla, a já jsem šel vždycky hned a rád.
}

\vers{3}{
Takhle na podzim, když už byl venku trochu chlad,\\
řek' jsem jestli smím už ji navždycky hřát,\\
ve svým objetí, jestli píseň jen jí tohle může dát?\\
\uv{No to je hotový,} řekla, a já už si ji měl tenkrát brát.
}

\vers{4}{
Jako malej kluk, ty dy dy dy dy,\\
teda zpívám si a všecko je mi fuk.\\
Da da da, di da du, di da du da du da,\\
da da, di da du, di da du da du da.
}

\vers{5}{
Přišel zimní čas, venku napadl sníh,\\
jí se trochu změnil hlas a mně zase zmrzl smích,\\
když jsem pochopil, že se opravdu už nemám čemu smát.\\
\uv{Ahoj, odcházím,} řekla, a já zůstal jako svatej stát.
}

\vers{6}{
Jeden jarní den, kdy jsem nemohl spát,\\
jako krásný sen, který měl se mi zdát,\\
přišla ke mně holka bezvadná a zůstala tu stát.\\
\uv{Já jsem Hedvika,} řekla, a já už ji měl rád.
}

\vers{7}{
Léto s Monikou bylo stoprocentní,\\
láskou velikou totiž hořel jsem k ní,\\
celý prázdniny měly nádhernej a trochu rychlej spád.\\
\uv{Pojď mě políbit,} řekla, a já jsem šel vždycky hned a rád.
}
\ns

\vers{8}{
Prostěž, a tak dál, přišel jinej muž,\\
co bych povídal, to přece znáte už\dots
}
