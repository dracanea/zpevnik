\hyt{touha}
\song{Touha}

\vers{1}{
\chord{F}Je to jak nic, to totiž neváží víc,\\
je to \chord{C}něco jak dech, jak přání na křídlech,\\
je to \chord{Dm}nemožnej cíl, kterej jsi \chord{A}vymyslil.
}

\vers{2}{
Je to jak sen, co ráno rozpustil den,\\
nic není dost velkej cíl, aby ses nepokusil,\\
volej k nebi přání a všechno bude k mání.
}
\vspace{10pt}

\nv Touha je \chord{F}zázrak, kámo, zázrak. \chord{C}\nc Touha je \chord{Dm}zázrak, kámo, zázrak. \chord{A}
\vspace{-10 pt}

\refrainn{1}{
\rep{Touha je \chord{F}zazrak, takže přej si co chceš, probouzej \chord{C}sílu, s ní už neklopýtneš,\\
touha jsou \chord{Dm}čáry, v hloubi duše je máš,\\
touha je \chord{A}zázrak, kámo, zázrak, vždyť už to znáš.}
}

\vers{3}{
Je to jak žár, pořádně velikej dar,\\
je to jak letící drak, naprosto přesnej prak.\\
Je to hukot moře, je to pramen v hoře.
}


\vers{4}{Touha je žít, zůstat a neodejít,\\
touha je padnout a vstát a rány nevnímat.\\
Je to tlukot srdce, je to poklad v ruce.
}


\vers{5}{Je to jak třpyt, jako sluneční svit\\
úžasná měsíční zář, je to jak Boží tvář.\\
Je to skrytá síla, je to zlatá žíla.
} \refsm{1}

\refrainn{2}{
Touha je krám, kde mají všechno a hned, vlak je tam k mání, i ten pitomej svět,\\
touha je brána ke všemu o čem jsi snil,\\
touha je cesta i cíl.
}
