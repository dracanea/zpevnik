\hyt{zasvoupravdoustat}
\song{Za svou pravdou stát}
\vers{1}{
Máš \chord{Am}všechny trumfy mládí a \chord{G}ruce čistý \chord{Am}máš,\\
jen \chord{Dm}na tobě teď \chord{Am}záleží, na \chord{E\7}jakou hru se \chord{Am}dáš.
}

\refrainn{1}{
\rep{Musíš \chord{Am}za svou pravdou \chord{E\7}stát,\\
musíš za svou pravdou \chord{Am}stát.}
}

\vers{2}{
Už víš, kolik co stojí, už víš, co bys rád měl,\\
už ocenil jsi kompromis a párkrát zapomněl.
}

\refrainn{2}{
\rep{Že máš za svou pravdou stát,\\
že máš za svou pravdou stát.}
}

\vers{3}{
Už nejsi žádný elév, co prvně do hry vpad',\\
už víš, jak s králem ustoupit a jak s ním dávat mat.
}

\refrain{3}{
\rep{Takhle za svou pravdou stát,\\
takhle za svou pravdou stát.}
}

\vers{4}{
Teď přichází tvá chvíle, teď nahrává ti čas,\\
tvůj sok poslušně neuhnul, a ty mu zlámeš vaz.
}

\refrain{4}{
\rep{Neměl za svou pravdou stát,\\
neměl za svou pravdou stát.}
}

\vers{5}{
Tvůj potomek ctí tátu, ty vštěpuješ mu rád\\
to heslo, které dobře znáš, z dob, kdy jsi býval mlád.
}

\vers{6}{
=\mm 1.
}
