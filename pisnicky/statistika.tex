\hyt{statistika}
\song{Statistika} \interpret{uhlirsverak}{Uhlíř \& Svěrák}
\sub{z pohádky Princové jsou na draka}

\vers{1}{
Je \chord{C}statisticky dokázáno,\\
že \chord{Am}slunce vyjde každé ráno\\
a \chord{C}i když je tma jako v ranci,\\
tak \chord{Am}noc už nemá žádnou šanci.
}

\refrain{
\rep{Statis\chord{F}tika nuda \chord{G}je, má však \chord{C}cenné úda\chord{Am}je,\\
nekle\chord{F}sejte na mys\chord{G}li, ona \chord{F}vám to vyčís\chord{C}lí.}
}

\vers{2}{
Když drak u sluje síru pouští\\
a Honza na něj číhá v houští,\\
tak statistika předpovídá,\\
že nestvůra už neposnídá.
} \refsm{}

\vers{3}{
Tak vyřiďte té ctěné sani,\\
že záleží to čistě na ní,\\
když nepustí ji choutky dračí,\\
tak bude o hlavičky kratší.
} \refsm{}
