\hyt{muzinejvicslusisolo}
\song{Muži nejvíc sluší sólo}
\sub{z muzikálu Noc na Karlštejně}

\vers{1}{
\chord{Am}Vlaje fábor, vlají vlajky, kde se ženský nepletou,\\
\chord{G}dělám stejně dobře krajky jako tečku za větou.
}

\vers{2}{
Dlouhý život v domácnosti připomíná requiem\\
za takových okolností nesmějem se, nepijem.
}

\refrain{
\chord{C}Muži nejvíc sluší sólo, v bitvě, jako \chord{G}u plotny,\\
ba i pidimuž je kolos, když je doma \chord{C}samotný.\\
La la la\dots
}

\vers{3}{
Nebudu vám říkat dlouze, co o ženských všechno vím,\\
svoje prádlo svěřím pouze zas a jenom rukám svým.
}

\vers{4}{
I ta husa radši vidí, když jí peří škube chlap,\\
kdo se za to, páni, stydí, ten je duchem stár a sláb.
} \refsm{}

\vers{5}{
To, co zvládnou ženský ruce, zvládne ruka kluka líp,\\
zamést, zašít, vyprat, upéct, a pak nabrousit si šíp.
}

\vers{6}{
Kam nemůže zajít ďábel, tam si ženskou dosadí,\\
ať jsi Kain a nebo Ábel, vyhýbej se podhradí!
}
