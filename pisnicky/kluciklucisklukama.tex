\hyt{kluciklucisklukama}
\song{Kluci kluci s klukama}

\vers{1}{
\chord{A}Kluci mají \chord{D}rádi \chord{A}míče, lítaj' s nima po par\chord{E\7}ku.\\
\chord{E\7}Holky, co se míče týče, \chord{H\7}dávaj' přednost kočár\chord{E}ku.
}

\vers{2}{
Kluci loví ryby z tůňky nebo lahve z Vltavy,\\
holky na to nemaj' buňky, lovení je nebaví.
}

\refrain{
\chord{E\7}A proto \chord{A}kluci, kluci s \chord{D}klukama, \chord{E}holky, holky s \chord{A}holkama.\\
\chord{A}Je to \chord{A\7}léty, \chord{D}je to \chord{A\dm}věkem, \chord{E}je to těma roka\chord{A}ma.
}

\vers{3}{
Než se léto s létem setká, hnedka je to jinačí,\\
do Karla se zblázní Bětka, Jaromír zas do Káči.
}

\vers{4}{
Přejdou deště aprílový, usadí se v kraji máj,\\
všechny holky kluky loví, najednou je zajímaj'.
} \refsm{}

\vers{5}{
Pánům vlasy náhle zřídnou, dámám do nich padne sníh,\\
sledujte to s tváří klidnou na hodinkách kapesních.
}

\vers{6}{
Děda s dědou v šenku hačaj', žejdlík s pěnou vábí je,\\
babičky si zajdou na čaj do kavárny Slávie.
} \refsm{}

\cod{
Protože je to léty, je to věkem, je to těma rokama.
}
