\song{Prosím tě, holka}

\intro{\textbf{\bas{G A H} C C\kk\dm G E\7 A\7 D\7}}

\vers{1}{
\chord{G}Prosím tě, holka, \chord{C\7}nemrač se tolik,\\
\chord{G}nebuď už se mnou \chord{E\7}na kordy.\\
\chord{C}Jsem jenom \chord{C\kk\dm}blázen \chord{G}a alko\chord{E\7}holik,\\
\chord{A\7}co umí čtyři \chord{D\kk}akor\chord{D}dy.

\vers{2}{
Akordy čtyři a písní stovku,\\
navíc pár príma historek.\\
Nalej si vodku, mně becherovku,\\
půjdem si spolu sednout na dvorek.
}

\refrain{
\chord{C\7}Měsíc, ten osamělý cestující časem,\\
\chord{G}decentně poodejde s vážnou tváří mima,\\
\chord{C\7}potom tě rozechvěju neumělým hlasem,\\
a \chord{G}když ne já, tak \chord{B\dm}určitě tě \chord{Am\7}rozechvěje \chord{D\7}zima.
}

\vers{3}{
Z oblohy cosi bílýho padá\\
a na tvým nose taje to.\\
Že já jsem starej a ty jsi mladá?\\
Aspoň mám něco najeto.
}
