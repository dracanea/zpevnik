\hyt{dvahavrani}
\song{Dva havrani}

\vers{1}{
Když jsem se z \chord{Dm}pole \chord{C}vrace\chord{Dm}la,\\
dva havra\chord{Dm}ny jsem \chord{C}slyše\chord{Dm}la,\\
jak jeden \chord{F}druhého se \chord{C}ptá:\\
\rep{\uv{\chord{Dm}kdo dneska veče\chord{C}ři nám \chord{Dm}dá?}}
}

\vers{2}{
Ten první k druhému se otočil\\
a černým křídlem cestu naznačil.\\
Krhavým zrakem k lesu hleděl,\\
\rep{a takto jemu odpověděl:}
}

\vers{3}{
\uv{Za starým náspem, v trávě schoulený,\\
tam leží rytíř v boji raněný\\
a nikdo neví, že umírá,\\
$\backslash$: jen jeho kůň a jeho milá.} :$\backslash$
}

\vers{4}{
\uv{Jeho kůň dávno po lesích běhá\\
a jeho milá už jiného má.\\
Už pro nás bude dosti místa,\\
$\backslash$: hostina naše už se chystá.} :$\backslash$
}

\vers{5}{
\uv{Na jeho bílé tváře usednem\\
a jeho modré oči vyklovem.\\
A až se masa nasytíme,\\
$\backslash$: z vlasů si hnízdo postavíme.} :$\backslash$
}
