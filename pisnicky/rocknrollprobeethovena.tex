\hyt{rocknrollprobeethovena}
\song{Rock'n'roll pro Beethovena}
\sub{z filmu Šakalí léta}

\vers{1}{
\chord{C\7}Včera v noci měl jsem divnej sen,\\
\chord{F\7}v tom snu ke mně přišel Beetho\chord{C\7}ven,\\
abych \chord{G\7}prej si s ním šel \chord{F\7}zahrát rock'n'\chord{C}roll.
}

\refrain{
Ó \chord{C\7}jé, rock'n'roll, ó jé, rock'n'roll,\\
ó \chord{F\7}jé, rock'n'roll, ó \chord{C\7}jé, rock'n'roll,\\
abych \chord{G\7}prej si sním šel \chord{F\7}zahrát rock'n'\chord{C}roll.
}

\vers{2}{
Já hned jak blázen poslech' jsem ten sen,\\
oblík' jsem se a šel za ním ven\\
a celou noc jsme hráli rock'n'roll.
} \refsm{} (a celou noc jsme hráli rock'n'roll)

\vers{3}{
Beethoven to asi kdysi hrál,\\
ten to uměl rozjet, až jsem řval,\\
jó, ten vám uměl rozjet rock'n'roll.
} \refsm{} (ten vám uměl rozjet rock'n'roll)

\vers{4}{
Přišlo ráno a já jsem už vstal,\\
kosti mi snad v těle rozlámal,\\
věděl jsem ale, jak hrát rock'n'roll.
} \refsm{} (věděl jsem ale, jak hrát rock'n'roll)
