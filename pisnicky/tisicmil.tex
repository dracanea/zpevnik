\hyt{tisicmil}
\song{Tisíc mil}
\vers{1}{
V nohách \chord{G}mám už tisíc \chord{Em}mil, stopy \chord{Am}déšť a vítr \chord{C}smyl\\
a můj \chord{Am}kůň i já jsme \chord{D\7}cestou znaveni.
}

\refrainn{1}{
Těch tisíc \chord{G}mil, těch tisíc \chord{Em}mil má jeden \chord{Am}směr a jeden \chord{C}cíl,\\
bílej \chord{Am}dům, to malý \chord{D\7}bílý stave\chord{G}ní.
}

\vers{2}{
Je tam stráň a příkrej sráz, modrá tůň a bobří hráz,\\
táta s mámou, který věřej dětskejm snům.
}

\refrainn{2}{
Těch tisíc mil, těch tisíc mil, má jeden směr a jeden cíl,\\
jeden cíl, ten starej, známej, bílej dům.
}

\vers{3}{
V nohách mám už tisíc mil, teď mi zbývá jen pár chvil,\\
cestu znám a ta se tam k nám nemění.
}\refsm{1}

\vers{4}{
Kousek dál a já to vím, uvidím už stoupat dým,\\
šikmej štít nad střechy ční k nebesům.
}\refsm{2}
