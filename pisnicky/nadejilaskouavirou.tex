\hyt{nadejilaskouavirou}
\song{Nadějí, láskou a vírou}
\hyl{oldtimereligion}{Old Time Religion}

\refrain{
Nadějí, \chord{G}láskou a vírou, nadějí \chord{D\7}láskou a \chord{G}vírou,\\
nadějí \chord{G\7}láskou a \chord{C}vírou,\chord{C\kk\dm}\mm\sm se \chord{G}píseň \chord{D\7}může \chord{D}stát.
}

\vers{1}{
Já když bejval ještě dítě, táta řek': \uv{Čas naučí tě!\\
Naději, že máš vždycky, dítě, a co víc si můžeš přát?}
} \refsm{}

\vers{2}{
Vím, že láska neumírá tomu, kdo ji neodpírá,\\
vím, že láska neumírá, tak neboj se mít rád.
} \refsm{}

\vers{3}{
Jenom ten, kdo v pravdu věří, rukou spravedlivou měří,\\
jenom gen, kdo v pravdu věří, ten může s námi hrát.
} \refsm{}

\vers{4}{
Dnes tvým krokům kámen brání, píseň tvá však nemá stání,\\
dnes tvým krokům kámen brání, pojď na cestu se dát!
} \refsm{}
