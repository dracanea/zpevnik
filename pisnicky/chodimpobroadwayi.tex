\song{Chodím po Broadwayi}

\vers{1}{
Chodím \chord{D}po Broadwayi hladov sem a tam,\\
chodím po Broadwayi hladov sem a \chord{A\7}tam,\\
chodím \chord{D}po Broad\chord{D\7}wayi, \chord{G}po Broad\chord{Gm}wayi,\\
\chord{D}po Broadwayi \chord{A\7}hladov sem a \chord{D}tam.
