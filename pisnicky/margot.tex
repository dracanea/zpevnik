\hyt{margot}
\song{Margot}

\refrain{
\rep{\chord{D}Margot češe na vinici v koši v koši v košili,\\
\chord{D}Margot češe na vinici \chord{A}jen \chord{D}tak.}
}

\vers{1}{
\chord{D}Kdopak to jde po silnici, \chord{A}Mar\chord{D}got?\\
\chord{D}Kdopak to jde po silnici, \chord{A}Mar\chord{D}got?\\
\chord{D}Šlapou si tu tři chasníci v koši v koši v košili\\
\chord{D}šlapou si tu tři chasníci \chord{A}jen \chord{D}tak.
}

\vers{2}{
Dej nám vína do čepice, Margot.\\
Dej nám vína do čepice, Margot.\\
Dám vám bičem na zadnice v koši v koši v košili,\\
dám vám bičem na zadnice jen tak.
} \refsm{}

\vers{3}{
Pročpak ztichla tvoje ústa, Margot?\\
Pročpak ztichla tvoje ústa, Margot?\\
Dva utekli, třetí zůstal, v koši v koši v košili,\\
dva utekli, třetí zůstal jen tak.
} \refsm{}

\vers{4}{
V sadě kvetou rozmarýnky, Margot.\\
V sadě kvetou rozmarýnky, Margot.\\
Margot brečí, pere plínky, v koši v koši v košili,\\
Margot brečí, pere plínky, jen tak.
} \refsm{}
