\hyt{psicenazrzla}
\song{Psice nazrzlá}

\vers{1}{
\chord{C}Jakej já byl samotář a \chord{F}lovec zkuše\chord{C}nej,\\
\chord{C}rád jsem nabíd' větru tvář, i \chord{D\7}když byl stude\chord{G\7}nej.\\
\chord{C}Srub jsem míval dřevěnej a \chord{F}psici nazrz\chord{C}lou,\\
u srubu schod kamennej a studnu zamrzlou.
}

\vers{2}{
A že pít se nedala ta voda zamrzlá,\\
řek' jsem, by mě čekala, ta psice nazrzlá.\\
\uv{Hlídej ten srub dřevěnej a sebe, jak by smet,\\
u schod kamennej a v týhle studni led.}
}

\vers{3}{
Pak jsem skočil na koně a jel do města pít,\\
večer v nóbl salóně si můžu dovolit.\\
Patří mi srub dřevěnej a psice nazrzlá,\\
u srubu schod kamennej a studna zamrzlá.
}

\vers{4}{
V městě bylo veselo a holek príma moc,\\
mně se domů nechtělo, tak pil jsem celou noc.\\
Propil jsem srub dřevěnej a psici nazrzlou,\\
u srubu schod kamennej a studnu zamrzlou.
}

\vers{5}{
Ale když dám dolar k dolaru a všechno na knížku,\\
řeknu holkám u baru, že chci si k ježíšku\\
pořídit srub dřevěnej a psici nazrzlou,\\
u srubu schod kamennej a studnu zamrzlou.
}
