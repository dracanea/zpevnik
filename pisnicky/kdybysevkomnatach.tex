\hyt{kdybysevkomnatach}
\song{Kdyby se v komnatách}
\sub{z pohádky Princové jsou na draka}

\vers{1}{
\chord{C}K životu na zámku \chord{Gm}mám jednu poznámku,\\
\chord{F}je tu neveselo, je tu \chord{C}truchlivo.\\
\chord{C}V ostatních královstvích \chord{Gm}nezní tak málo smích,\\
\chord{F}není neveselo, není \chord{C}truchlivo.
}

\refrain{
\chord{C\7}Kdyby se v \chord{F}komnatách běhoun jak \chord{Em}hrom natáh'\\
a na něm \chord{Dm}akrobati \chord{G}začali \chord{C}kejklovati.\\
\chord{C\7}Kdyby nám v \chord{F}paláci pištěli \chord{Em}dudáci,\\
to by se \chord{Dm}krásně žilo, \chord{G}to by bylo \chord{C}ráj.
}

\vers{2}{
Kde není muzika, tam duše naříká,\\
tam je neveselo, tam je truchlivo.\\
Chtěla bych dvůr pestrý, kde znějí orchestry,\\
kde není neveselo, žádné truchlivo.
} \refsm{}
