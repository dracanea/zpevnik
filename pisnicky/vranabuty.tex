\hyt{vranabuty}
\song{Vrána} \interpret{buty}{Buty}
\sub{Buty}

\vers{1}{
\textbf{C Am Dm G}\\
Letí vrána, letí vzduchem, nepospíchá, šetří síly.\\
Jedním okem, jedním uchem sleduje, co ostatní vrány.
}

\vers{2}{
\textbf{D Hm Em A}\\
Taky letí, taky letí, druhá vrána, třetí vrána.\\
Nespěchají, šetří síly, ještě dlouho poletí vzduchem.
}

\vers{3}{
\textbf{E C\kk m F\kk m H}\\
Letí vrány, letí, letí, nepotí se, vzduch je chladí.\\
Letí dlouho, všechny spolu, jedna vrána jako druhá.
}

\vers{4}{
\textbf{F\kk D\kk m G\kk m C\kk}\\
Na chvíli se vystřídají, první vrána už se cítí\\
unavená, unavená, tak se schová na konec hejna.
}

\bridge{
\textbf{A\bb Fm Bm E\bb}\\
Letí, letí, letí, letí\dots\\
\vinv
\textbf{B Gm Cm F}\\
Letí, letí, letí, letí\dots
}

\vers{5}{
\textbf{C Am Dm G}\\
Letí vrány, letí, letí, ještě pořád letí vzduchem.\\
Nic si spolu neříkají, unavené černé vrány.
}

\cod{
\textbf{Em Am Dm G C}\\
Až si najdou dobré místo, sednou si a budou klidně spát.
}
