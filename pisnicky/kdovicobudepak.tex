\hyt{kdovicobudepak}
\song{Kdo ví, co bude pak}

\vers{1}{
Kdo \chord{C}ví, co bude \chord{Am}pak, \chord{Dm}až otevře \chord{G}prvně oči,\\
kdo \chord{C}ví, co bude \chord{Am}pak, až \chord{Dm}otevře je \chord{G}víc?\\
\rep{Zda \chord{F}najde cestu \chord{G}svou, nebo \chord{C}cizí, dláždě\chord{Am}nou\\
a \chord{F}v davu půjde \chord{G}si, či \chord{F}v čele \chord{G}proce\chord{C}sí?} 
}

\vers{2}{
Kdo ví, co bude pak, zda podrží si ideály,\\
kdo ví, co bude pak, zda si je udrží?\\
\rep{Co bude vlastně zač, zbabělec nebo rváč,\\
chladný jako sníh, či hořet jako vích?}
}

\vers{3}{
Kdo ví, co bude pak, až hlavou vrazí prvně do zdi,\\
kdo ví, co bude pak, až prvně narazí?\\
\rep{Zda se jen otřese a nebo ohne se,\\
zda nebude se bát si vždycky za svým stát?}
}

\vers{4}{
Kdo ví, co bude pak, zda-li ho lež nezdolá,\\
kdo ví, co bude pak, jestli ho nesrazí?\\
\rep{Bude mít zelenou, nebo žít na kolenou\\
a zda-li bude svět za pětadvacet let?}
}

\vers{5}{
Kdo ví, co bude pak, zda-li taky ptát se bude,\\
kdo ví, co bude pak, jestli se bude ptát?\\
\rep{Zda ještě čekat vlak a nebo jít jen tak,\\
ať slunce nebo mrak, kdo ví, co bude pak?}
}
