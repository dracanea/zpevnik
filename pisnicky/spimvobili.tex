\hyt{spimvobili}
\song{Spím v obilí} \interpret{tucny}{Michal Tučný}

\refrainn{1}{
\chord{D}Ó, ladylí, ještě \chord{G}mám fůru \chord{D}času,\\
\chord{G}mám fůru \chord{D}času, a tak \chord{A}spím v obilí\\
a \chord{D}zdá se mi sen plný \chord{G}dívek a \chord{D}dětí,\\
\chord{G}dívek a \chord{D}dětí a je \chord{A}líný letní \chord{D}den.
}

\vers{1}{
Kam půjdu dál, až se probudím a vstanu,\\
probudím a vstanu, na to ještě nemyslím.\\
Já spím v obilí a zdá se mi právě,\\
sen, jak přímo z lahve starou režnou popíjím.
}

\vnv

\note{pomalu}

\vers{2}{
Za chvíli vstanu, mám ještě stihnout kdesi cosi,\\
slyším zvonit kosy, už mi kosí obilí.\\
Já vím, strniště jak nůž do zad tlačí,\\
děkuju, to stačí, vzbudili mě právě včas.
}
