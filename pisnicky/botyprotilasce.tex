\hyt{botyprotilasce}
\song{Boty proti lásce}

\vers{1}{
\chord{E}Zas mi říkal, že má něco pro mne,\\
a to něco, že prý láska je.\\
\chord{A}Já však nechci žádnou lásku v domě, je,\\
\chord{E}přináší jen žal a výdaje.\\
Mám \chord{G}proti lásce \chord{E}boty, ty \chord{G}chrání paní \chord{E}svou,\\
ty \chord{G}boty vždycky \chord{E}jdou a všechno hezké pošla\chord{\bas{E E D\kk D\kk D D C\kk C\kk C C H H A\kk A G\kk G E}}pou
}

\vers{2}{
Žiju sama díky těmto botám,\\
žiju sama zásluhou těch bot.\\
Často sice klukům hlavy motám, je,\\
botám ale city nejsou vhod.\\
V těch botách ráda šlapu, zvlášť po lásce nás dvou,\\
ty boty vždycky jdou a všechno hezké pošlapou.
}

\vers{3}{
Tyhle boty nosím někde v duši,\\
někde v duši, ne však na nohou.\\
Jsou tam dole, kde mi srdce buší, je,\\
od bušení mu však pomohou.\\
V těch botách ráda šlapu, a to se mi prý mstí,\\
prý v těchto botách jdu, a tak si šlapu po štěstí.
}
\vnv

\nv\emph{rec:} Pozor, boty! Pochodem v chod!
