\hyt{barbarskepobrezi}
\song{Barbarské pobřeží} \interpret{asonance}{Asonance}
\ns
\vspace{-10pt}

\vers{1}{
\chord{Dm}Dvacet stupňů levobok a držte \chord{C}jižní \chord{Dm}směr!\\
Máme \chord{F}Gibraltar v \chord{C}zádech, a když \chord{B}vítr vydr\chord{A}ží,\\
snad \chord{Dm}uvidíme \chord{C}Afriku už \chord{Dm}zítra nave\chord{Am}čer,\\
náš \chord{B}cíl je přístav Oran na bar\chord{Am}barském pobře\chord{Dm}ží.
}

\vers{2}{
\uv{Vidím loď, velkou loď!,} křičí hlídka z ráhnoví,\\
máme Gibraltar\dots\\
když galéra se na obzoru v dálce objeví,\\
náš cíl je\dots
}

\vers{3}{
Jsi pirát, jsi lupič, nebo válečná loď snad?\\
žádnou vlajku nemáš, jak tě máme přivítat?
}

\vers{4}{
To ne, piráti nejsme, my jsme loď obchodní,\\
do Liverpoolu plujeme už pětadvacet dní.
}

\vers{5}{
Tak spustili jsme člun, on ve vlnách už je.\\
\uv{Vemte naši poštu domů do rodné Anglie!}
}

\vers{6}{
Však náhle vzduchem zahřměly dvě salvy z jejich děl\\
a černý prapor pirátů na stěžeň vyletěl.
}

\vers{7}{
Jejich útok byl tak rychlý, že všechny překvapil,\\
ale pak náš výstřel z děla jim stěžeň ustřelil.
}

\vers{8}{
\uv{Slitování mějte!,} ty krysy křičely,\\
tak dali jsme jim milost, ke dnu je poslali.
}

\vers{9}{
S piráty jsme se rvali, bili se dvojnásob,\\
loď byla jejich rakví a moře jejich hrob.
}
