\hyt{carminaamorosa}
\song{Carmina amorosa}

\vers{1}{
\rep{\chord{E}More festi querimus \chord{A}virum virtuosum,\\
\chord{E}quem et benedicimus hos\chord{A}pitem \chord{E}glori\chord{A}osum,}\\
\chord{Hm}et ab ipso \chord{E}petinus, \chord{A}munus graciosum,\\
ut \chord{Hm}sumam nobis \chord{E}conferat \chord{A}trium \chord{E}soli\chord{A}dorum.
}

\vers{2}{
\rep{Kdo svou duši spravuje tam, kde korbel cinká,\\
ten svou milou miluje, až když už Pán Bůh spinká.}\\
Láska hoří kolem nás jako plamen svíce,\\
až ten oheň spolkne čas, nebude jí více.
}

\vers{3}{
\rep{Kdo lže kněžím v kostele, čert ho stáhne z kůže,\\
než ti milá ustele, však trochu lhát se může.}\\
Láska hoří kolem nás jako plamen svíce,\\
až ten oheň spolkne čas, nebude jí více.
}

\vers{4}{
\rep{Neptejte se učenců na elixír mládí,\\
zeptejte se mládenců, těch, kteří panny svádí.}\\
Láska letí kolem nás jako z nebe kámen,\\
chyť ji do svých rukou včas, sic je s tebou amen.
}

\cod{\chord{E}Sic je s tebou \chord{A}amen.}
