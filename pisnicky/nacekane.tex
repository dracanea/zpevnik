\hyt{nacekane}
\song{Na čekané} \interpret{spiritualkvintet}{Spirituál kvintet}

\vers{1}{
\chord{A}Maličký \chord{D}pokojík \chord{A}ve starém \chord{D}domě,\\
\chord{A}suterén \chord{D}s okýnkem \chord{A}a kočky \chord{C\kk m}potmě\\
\chord{G}chodily \chord{D}po schodech \chord{A}za tebou \chord{D}dolů,\\
\chord{A}potichu \chord{Hm\7}čekaly \chord{D}na zbytky \chord{A}stolu. \chord{A\sus{4}}
}

\vers{2}{
Maličké obrázky na zdech tam visely,\\
sluneční paprsky jen k ránu svítily\\
a čísi prsty na okno klepaly,\\
jakoby drápky na dveře škrábaly.
}

\refrainn{1}{
\chord{C\kk m}Mlsně tam \chord{Cm}čekaly, \chord{Hm}zavřít se \chord{Bm}nedaly,\\
\chord{A}vždycky se \chord{Hm\7}vracely \chord{D}tam k tobě \chord{A}dolů. \chord{A\sus{4}}\\
\chord{C\kk m}Na dveře \chord{Cm}škrábaly, \chord{Hm}na okno \chord{Bm}klepaly,\\
\chord{A}vždycky se \chord{Hm\7}vracely \chord{D}tam k tobě \chord{A}dolů, tam k tobě dolů. \chord{A\sus{4}}
}

\vers{3}{
O pár let později už v novém domě,\\
zvýšené přízemí a záclony v okně.\\
Po schodech nahoru kočky už nechodí,\\
v neděli večer u tebe neloudí.
}

\refrainn{2}{
Asi hlad nemají, snad cestu neznají,\\
možná se vracejí tam k tobě dolů.
}

\vers{4}{
Zas stejné obrázky na zdech tam visí,\\
sluneční paprsky i k večeru svítí.\\
Jenže okno je vysoko, tak nikdo neklepe,\\
drápky už neškrábou a láska nekvete, jó nekvete.
}
