\vers{1}{
\chord{E}Já harfu mám, ty harfu máš, dal všem svým \chord{A}dětem harfu \chord{H\7}Pán,\\
až \chord{E}vydáme se \chord{E\7}do ráje, pochod \chord{A}má harfa \chord{A\7}zahraje\\

\vers{2}{
Já křídla mám, ty křídla máš, dal všem svým dětem křídla Pán,\\

já polehoučku jako dým na křídlech poletím nebem, \emph{(já to vím)} nebem,\\

}

\vers{3}{
Já sílu mám, ty sílu máš, dal všem svým dětem sílu Pán,\\
v tem velký den hned po ránu pěkně zabouchám na bránu,\\
to musí znít celým nebem, \emph{(já to vím)} nebem, \emph{(já to vím)} nebem,\\
ránu já mám parádní, tu poznáš, až zazní nebem, \emph{(já to vím)} nebem,\\
věřte mi, zabouchám celým nebem!
}

\vers{4}{
Já píseň znám, ty píseň znáš, dal všem svým dětem píseň Pán,\\
já vím, že jednou přijde čas, Tvůj na kůr mě pozve hlas,\\
já zazpívám celým nebem, \emph{(já to vím)} nebem, \emph{(já to vím)} nebem,\\
já všechny písně, které znám, jednou zazpívám nebem, \emph{(já to vím)} nebem,\\
věřte mi, zazpívám celým nebem!
}
