\hyt{kayandra}
\song{Kayandra} \interpret{spiritualkvintet}{Spirituál kvintet}

\vers{1}{
\chord{A}Tam v přístavu na haldě \chord{D}rybářskejch \chord{A}sítí\\
večer co večer vždy \chord{G}stařičkej \chord{E}Ben\\
hledí \chord{A}do dálky k ostrovům, kde maják svítí,\\
poslouchá moře a \chord{G}sní krásnej \chord{A}sen.
}

\refrain{
Tou starou lodí zas plout do Kayandry,\\
látat plachty a popíjet rum,\\
potom houpat se v koši a s nákladem ambry\\
z dálky už vyhlížet svůj rodnej dům.
}

\vers{2}{
V sousední vsi míval za mlada milou,\\
oči jak uhel, vlasy jak len,\\
její táta měl statek a lán lesů s pilou,\\
pro něj byl jen chudej námořník Ben.
} \refsm{}

\vers{3}{
Jednou, když domů se z moře zas vrátil,\\
v kabátě prstýnek pro milou Ann,\\
slyšel zvony, jak hlásí, že svý štěstí ztratil,\\
našel se ženich, co víc měl než Ben.
} \refsm{}

\vers{4}{
Stříbrnej prstýnek na slunci svítí,\\
víckrát ho nikomu nenabíd' Ben,\\
teď tu sedí a slanej déšť příboje cítí,\\
poslouchá moře a sní krásnej sen.
} \refsm{}
