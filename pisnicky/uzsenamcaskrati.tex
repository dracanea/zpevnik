\hyt{uzsenamcaskrati}
\song{Už se nám čas krátí}

\vers{1}{
\chord{Am}Kde je, kde je \chord{C}sláva Říma, \chord{G}kde je, kde je \chord{C}sláva \chord{C\7}Říma?\\
V \chord{F}dávných troskách \chord{Am}dávno \chord{Em}dřímá, \chord{Am}tam, \chord{C}kde i světlo \chord{G}mizí \chord{C}v tmách. \chord{E\7}
}

\vers{2}{
\rep{Kudy bloudí dávná slova?}\\
Zjeví se a ztratí znova, tam, kde i světlo mizí v tmách. 
}

\refrain{
\rep{Už se, už se nám čas krátí,}\\
přijde den a půjdem spáti, tam, kde i světlo mizí v tmách.
}

\vers{3}{
\rep{Kam se, kam se ztratí činy?}\\
Mrtvé činy z naší viny -- tam, kde i světlo mizí v tmách.
}

\vers{4}{
\rep{Kam se, kam se ztrácí touha?}\\
Únava nám zbývá pouhá tam, kde i světlo mizí v tmách.
}

\refsme{}

\vers{5}{
\rep{Kde jsou, kde jsou velké lásky?}\\
Půvabné jak semikrásky -- tam, kde i světlo mizí v tmách.
}

\vers{6}{
\rep{Kde se, kde se skrývá pravda?}\\
Pálená i mrazem zvadlá, dál vábně voní věčnou tmou.
}
