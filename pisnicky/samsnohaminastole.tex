\hyt{samsnohaminastole}
\song{Sám s nohami na stole}

\vers{1}{
\chord{G}Zas vodovod slzí a žiarovka žmurká,\\
to \chord{Hm}zbožňované slnko múch.\\
Aj \chord{C}my sa popálime o všetky svoje slnká\\
a \chord{Am}vydrží len \chord{Cm}dobro\chord{D}druh.
}

\vers{2}{
Dám si nohy na stôl a dolejem víno,\\
nech dostanem to skôr do žíl,\\
že keby som tam zomrel, rýchlo by mi došlo, ako som si pekne žil.
}

\refrain{
\chord{G}Sám s nohami \chord{Am}na stole, zrazu \chord{C}život ma prebo\chord{G}lel.\\
\chord{G}Sám s nohami \chord{Am}na stole, zrazu \chord{C}dobré je aj to \chord{G}zlé,\\
a \chord{Hm}to mi robí celkom \chord{G}fajn, znova \chord{Cm}na seba sa \chord{Am}podo\chord{D}bám.
}

\vers{3}{
Či peňaženku nosíš v saku pri srdci,\\
či v zadnom vrecku nohavic,\\
svet je pekný, iba pokiaľ si na svete,\\
zvyšok doplácaš a nemáš nič.
}
