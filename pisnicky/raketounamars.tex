\hyt{raketounamars}
\song{Raketou na Mars}

\vers{1}{
\chord{C}Raketou na Mars poletíme zas ty a \chord{G}já.\\
\chord{Dm}Raketou jen já a ty, \chord{G}a to touhle tou jen \chord{C}pro nás.\\
\chord{C}Raketou na Mars poletíme \chord{C\7}zas, no tak\\
\chord{F}pěkný počasí, to se ví, chtěj mít. Dou\chord{Dm}fej,\\
že ves\chord{C}mír neznámý uvi\chord{G}dí, kdo jsme my, já a \chord{C}ty. 
}

\vers{2}{
Raketou na Mars poletíme zas já a ty.\\
Raketou jen já a ty, všechny hvězdy jsou jen pro nás.\\
Raketou na Mars tobě pro radost a jak\\
láska ve mně plá a plá a plápolá\\
tou rourou z ocele letíme přátelé ke hvězdám.
}

\vers{3}{
Wo o o o o\dots\\
Kdo tu sílu má, milovat jak já, ach ta\\
láska ve mně plá a plá a plápolá,\\
láska nás povznáší, láska nás vynáší ke hvězdám.
}

\vers{4}{
Wo o o o o\dots\\
Raketou na Mars. Wo o o o o\dots
}
