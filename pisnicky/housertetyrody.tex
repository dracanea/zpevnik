\hyt{housertetyrody}
\song{Houser tety Rody}

\vers{1}{
\chord{C}Jdou to říct Rody,\\
\chord{G}naší tetě \chord{C}Rody, všichni\\
\chord{C}jdou \chord{G}to říct \chord{Am} Rody,\\
že ten \chord{G}starej houser \chord{C}chcíp'.
}

\vers{2}{
\rep{Plave hlavou dolů,}\\
bejt živej plave líp.
}

\vers{3}{
\rep{Housátka kejhaj,}\\
že se táta utopil.
}

\vers{4}{
\rep{V náhonu u mlejna,}\\
odkud vodu právě pil
}

\vers{5}{
\rep{Stará husa pláče,}\\
s h(H)usákem, že bylo líp.
}

\vers{6}{
Jdou to říct Rody,\\
naší tetě Rody, všichni\\
jdou to říct Rody,\\
že ten starej houser chcíp.
}
