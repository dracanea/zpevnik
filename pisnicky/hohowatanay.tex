\song{Ho ho Watanay}
\sub{indiánská ukolébavka}

\vers{1}{
\chord{D}Spinkej můj maličký, \chord{C}máš v očích \chord{D}hvězdičky,\\
dám ti je \chord{C}do vlasů, tak \chord{G}usínej, tak \chord{D}usínej.
}

\refrain{
\chord{D}Ho ho Watanay, \chord{C}ho ho \chord{D}Watanay,\\
ho ho \chord{C}Watanay, Ki\chord{G}okena, Ki\chord{D}okena.
}

\vers{2}{
Sladkou vůni nese ti noční motýl z perleti,\\
vánek ho kolíbá, už usíná, už usíná.
}\refsm{}

\vers{3}{
V lukách to zavoní, rád jezdíš na koni,\\
má barvu havraní, jak uhání, jak uhání.
}\refsm{}

\vers{4}{
V dlani motýl usíná, hvězdička už zhasíná,\\
vánek, co ji k tobě nes', až do léta ti odlétá.
}\refsm{}
