\hyt{kovbojuvnarek}
\song{Kovbojův nářek} \interpret{spiritualkvintet}{Spirituál kvintet}

\vers{1}{
Když \chord{C}v Laredu \chord{Em}tenkrát jsem \chord{C}v ulicích \chord{G}brouzdal,\\
když \chord{C}v Laredu \chord{Dm}tenkrát jsem \chord{C}poprvé \chord{G}byl,\\
já \chord{C}kovboje \chord{Em}zahlíd', byl \chord{C}přikrytej \chord{G}plátnem,\\
tím \chord{Am}běloučkým \chord{Dm}plátnem, co \chord{G}studí jak \chord{C}jíl.
}

\vers{2}{
Ať píšťaly kvílí a bubny v tu chvíli\\
ať zadrnčí v pohřebním průvodě mým.\\
Tam v údolí u nás mě přikrejte hlínou,\\
jsem mladičkej kovboj a chybil jsem, vím.
}

\vers{3}{
Šest hazardních hráčů ať vezme mou rakev,\\
šest kovbojů zazpívá nad hrobem mým.\\
A vemte mě na hřbitov, zahažte hlínou,\\
jsem kovboj a hodně jsem chybil, já vím.
}

\vers{4}{
Šest veselých kovbojů ponese rakev,\\
šest nejhezčích dívek mě doprovodí.\\
A kytici růží mi na rakev dejte,\\
ať hroudy tak prudce mě neuhodí.
}
