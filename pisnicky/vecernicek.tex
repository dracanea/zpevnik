\hyt{vecernicek}
\song{Večerníček} \interpret{uhlirsverak}{Uhlíř \& Svěrák}

\vers{1}{
\chord{C}Umí jenom čtyři slova, \chord{F}dobrý večer, dobrou noc,\\
\chord{G}každý den je říká znova, \chord{C}nenamluví toho moc.\\
\chord{C}Koně řídí, auto řídí, \chord{F}jezdí na kole taky,\\
\chord{G}je tu pro nejmenší lidi, \chord{F}rozdává jim \chord{G}letá\chord{C}ky.
}

\refrain{
\chord{C}malí \chord{F}dopředu, \chord{G}větší \chord{C}dozadu,\\
\chord{Am}ruce \chord{F}pod bradu, \chord{G}přijíždí k nám Večerníček.\\
\chord{C}Všichni \chord{F}najednou, \chord{G}tiše \chord{C}usednou,\\
\chord{Am}přímo \chord{F}před bednou, \chord{G}přijíždí k nám Večerní\chord{C}ček.
}

\vers{2}{
Když hodiny bijou sedmou, malé děti v pyžamech,\\
ochotně si k němu sednou mezi kapradí a mech.\\
Čepici má papírovou, letos, jako předloni,\\
pohádku vám poví novou, a po ní se ukloní.
} \refsm{}

\cod{
Přijíždí k nám Večerníček.
}
