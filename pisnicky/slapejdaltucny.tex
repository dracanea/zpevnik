\hyt{slapejdaltucny}
\song{Šlapej dál}
\sub{Michal Tučný}
\vers{1}{
Hej, \chord{F}nandej na sebe modrý \chord{C\7}džíny,\\
vlak houká, v kopci je poma\chord{F}lej,\\
tak vstávej, z tý udusaný \chord{C\7}hlíny,\\
hej, tuláku, už je ráno, a tak oči otví\chord{F}rej.
}

\refrain{
Šlapej \chord{F}dál a táhni ke všem \chord{C\7}čertům,\\
tohleto je město prokla\chord{F}tý\dots\\
a šerif náš, ten nerozumí \chord{C\7}žertům,\\
a tak tohle ráno mohlo by bejt pro tebe dost \chord{F}zlý.
}

\vers{2}{
Hej, plandej už dál, jak vede cesta,\\
a koukej si chytit nějakej vlak.\\
Šerif náš je z pepřenýho těsta\\
a jak zmerčí tuláka, tak začne řádit jako drak.
}\refsm{}

\vers{3}{
Hej \chord{G}nandej na sebe modrý \chord{D\7}džíny,\\
vlak houká, v kopci je poma\chord{G}lej.\\
Jeď dál, kam povedou tě \chord{D\7}šíny,\\
a k našemu městu už se nikdy nevra\chord{G}cej.
}\refsm{}
