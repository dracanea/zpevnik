\hyt{salome}
\song{Salome} \interpret{kryl}{Karel Kryl}

\vers{1}{
\chord{Em}Něžná i proradná, \chord{G}krutá i bezradná,\\
\chord{D}plamen i červánek, ďábel i beránek, \chord{H}cukr i sůl.\\
\chord{Em}U vůně hřebíčku, \chord{G}u rytmu střevíčků\\
\chord{D}císař dnes myslí byl, za tanec přislíbil \chord{H}království půl.
}

\refrainn{1}{
\chord{Em}Salome, \chord{H}\nc\chord{E}noc už je na sklonku, \chord{G}Salome, podobnas' úponku,\\
\chord{D}podobna kytaře pro svého vladaře, \chord{H}Salome, tančíš.\\
\vinv
Salome, sťali už Křtitele, Salome, usměj se vesele,\\
točíš se ve víru, ústa, jak upíru krví ti planou, Salome\dots la la \chord{Em}la \sm\chord{G} \mm\chord{C} \mm\chord{H} \mm\chord{Em} \mm\chord{G} \mm\chord{H}
}

\vers{2}{
Noci už ubývá, císař se usmívá,\\
pokojně mohu žít, všeho lze použít pro dobrý stát.\\
Možná ho napadlo prastaré říkadlo: dějiny tvoří se, pro hlavy na míse nemá se štkát.
}

\refrainn{2}{
Salome, netanči, nechceš-li, Salome, hosté už odešli,\\
jenom roj komárů dopíjí z pohárů krůpěje vína.\\
\vinv
Salome, trochu jsi pobledla, Salome, v koutku jsi usedla,\\
víčka máš šedivá, nikdo se nedívá, Salome! Pláčeš?
}
