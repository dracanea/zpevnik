\hyt{codelastodelejrad}
\song{Co děláš, to dělej rád}

\vers{1}{
\chord{C}Můj táta byl jen střízlík, měl sílu nevel\chord{G\7}kou,\\
ale jak jsem ho znal, byl větší než král,\\
jenom jedinou myšlen\chord{C}kou.
}

\refrain{
\chord{C}Nesmíš být nikdy líný, nesmíš se práce \chord{G\7}bát,\\
jen se snaž, jak nejlíp to znáš, co děláš, to dělej \chord{C}rád.
}

\vers{2}{
Když jsem se někdy díval, jak mámu objímal,\\
tak natáhnul krk a lišácky mrk',\\
a nato hned mi povídal:
}

\vers{3}{
Ač z tváří mu úsměv zářil, někdy taky smůlu měl,\\
pak zaplakal, a když jsem se ptal,\\
řek' mi, \uv{Proč bych nebrečel?}
}

\vers{4}{
Vidím ten den jak včera, když nám horečkou zbled,\\
ale dříve než zhas', stačil pozvednou hlas,\\
a pak mi řek' naposled:
}
