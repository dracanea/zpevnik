\hyt{melanaocichbryle}
\song{Měla na očích brýle} \interpret{zich}{Karel Zich}
\ns
\note{capo 2}


\refrainn{1}{
\chord{A}Já ji \chord{Hm}znám, měla na očích brýle, \chord{A}ctím\\
půvab té chvíle, stála \chord{Hm}tam,\\
v očích vzdálené cíle, \chord{D}tak ji z léta \chord{A}znám.
}

\vers{1}{
\chord{E}Jel jsem právě sám, v autě místo mám,\\
\chord{D}tam se jedna dívka vejde,\\
\chord{E}zastavil jsem s chutí, \chord{D}já mám letní \chord{E}dívky \chord{A}rád.\\
Říkala \chord{E}jeď dál a přidej plyn, \chord{D}já na to, že víc to nejde,\\
\chord{E}moje auto jezdí \chord{D}s větrem v zádech \chord{E}šede\chord{A}sát.
} \refsm{1}

\refrainn{2}{
Já ji znám, jako jedinou z mála, vím,\\
na rohu stála, já byl sám,\\
ona se smála, tak ji z léta znám.
}

\vers{2}{
Dneska, den, co den, jezdím z města ven,\\
na rohu je místo prázdné,\\
do sobotních novin dávám pro ni inzerát.\\
Pak jezdím sem a tam a přidám plyn\\
a sám se divím, kde jen to vázne,\\
proč neví, že mám ji od loňského léta rád?
} \refsm{1} \refsm{2}

\refrainn{3}{
Stála \chord{C\kk m}tam, měla na očích brýle, \chord{H}ctím\\
půvab té chvíle, já ji \chord{C\kk m}znám,\\
v očích vzdálené cíle, \chord{E}tak ji z léta \chord{H}znám.
}
\ns

\refrainn{4}{
Já, já, já ji znám, v očích vzdálené cíle, vím,\\
stačila chvíle, vzpomínám,\\
měla na očích brýle, tak ji z léta znám.
}
