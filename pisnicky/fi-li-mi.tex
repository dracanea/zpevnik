\hyt{fi-li-mi}
\song{Fi-li-mi} \interpret{spiritualkvintet}{Spirituál kvintet}
\note{capo 1}

\vers{1}{
\chord{Em}Čert aby vzal už tuhle trať, kdo \chord{G}hledáš práci, tak se ztrať,\\
\chord{Em}že nemáš prachy, no tak ať, jó, tak se \chord{Hm}na to \chord{Em}dívám!
}

\refrain{
\chord{Em}Fi-li-mi-jó-ri-jú-ri-ej, \chord{G}fi-li-mi-jó-ri-jú-ri-ej,\\
\chord{Em}fi-li-mi-jó-ri-jú-ri-ej, vo tom \chord{Hm}si teď \chord{Em}zpívám.
}

\vers{2}{
Jen pražec chop a kolej suň, chyť lano, táhni jako kůň,\\
pod tíhou jako medvěd fuň, jó, tak se na to dívám!
} \refsm{}

\vers{3}{
Z kůže se loupeš jako had, je vedro, že by jeden pad',\\
na vodu smíš jen vzpomínat, jó, tak se na to dívám!
}

\vers{4}{
Když konečně máš vody dost, určitě přes ni stavíš most,\\
kláda ti ráda zlomí kost, jó, tak se na to dívám!
} \refsm{}

\vers{5}{
Na rukách už jsem potěžkal většinu těch okolních skal,\\
ještě to cejtí každej sval, jó, tak se na to dívám! 
} \refsm{}

\vers{6}{
Slunce už dělá z trávy troud, jen kdybych se směl vodsaď hnout,\\
na tuhle trať zapomenout, jó, tak se na to dívám!
} \refsm{}

\vers{7}{
\chord{Em}Jen Bůh mi víru zachovej a \chord{G}nasednout mi sílu dej,\\
můj \chord{Em}vagón bude \chord{C}pérovej, jó, \chord{Am}vo tom \chord{Hm}si teď \chord{Em}zpívám!
}

\refrain{
Fi-li-mi-jó-ri-jú-ri-ej, fi-li-mi-jó-ri-jú-ri-ej,\\
fi-li-mi-fi-li-mi-jú-ri-ej, vo tom si teď zpívám.
}

\cod{Vo tom si teď zpívám.}
