\hyt{slapejdalspiritual}
\song{Šlapej dál}
\sub{Spirituál kvintet}
\vers{1}{
Když svý \chord{Am}\mm černý tělo \chord{Dm}necej\chord{Am}tíš,\\
máš týden \chord{Am}hlad, žízeň \chord{Dm}v blátě \chord{Am}spíš,\\
musíš \chord{Am}jít, jen to \chord{Dm\7}nevzdá\chord{Am}vej, tak \chord{E\7}šlapej \chord{Am}dál. (\chord{E\7}\emph{Šlapej} \chord{Am}\emph{dál!})
}

\refrainn{1}{
Jen \chord{E}dál, (\emph{jen dál}) jen \chord{Am}dál!\\
Na zádech v ranci \chord{Am}máš jen \chord{Dm}bídu svou,\\
tak jen \chord{E\add{7}}šlapej \chord{Am}dál. \chord{E\add{7}}Šlapej \chord{Am}dál!
}

\vers{2}{
Když máš dřít na šlichtu mizernou,\\
svý známý v lochu, holku nevěrnou,\\
můžeš klít, jen to nevzdávej, šlapej dál.
}\refsm{1}

\vers{3}{
Když máš pech toho, kdo nekejvá,\\
tvejch slavnejch kamarádů ubejvá,\\
pak sám víš, těm už nescházíš, šlapej dál.
}\refsm{1}

\vers{4}{
Je nás víc, nemáme zastání,\\
jen špatnou barvu, špatný vyznání,\\
černej mrak, který narůstá, šlapej dál.
}

\refrainn{2}{
Jen dál, jen dál!\\
Na zádech v ranci máš i víru svou,\\
tak s ní šlapej dál. Šlapej dál!
}
