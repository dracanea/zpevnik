\hyt{zapomnetliva}
\song{Zapomnětlivá} \interpret{uhlirsverak}{Uhlíř \& Svěrák}

\vers{1}{
\textbf{A:} \chord{Am}Kde to je? Teď před chvílí jsem \chord{E\7}to tu ještě \chord{Am}měl.\\
\textbf{B:} \chord{Am}Co hledáš, švagře milý? Možná, že \chord{E\7}bych vě\chord{Am}děl.\\
\textbf{A:} \rep{\chord{C}Takovej ten, s \chord{G}takovou tou, \chord{F}jak to jen říci \chord{C}jednou větou?\\
\hspace*{20pt}\chord{C}Taková ta \chord{G}s takovým tím.\\
\textbf{B:} \chord{F}Bohužel, švagře \chord{E\7}nerozu\chord{Am}mím.}
}

\vers{2}{
\textbf{A:} No tohle, to jsem blázen, to jsou mi problémy.\\
\textbf{B:} Možná to spadlo na zem a je to na zemi.\\
\textbf{A:} \rep{Na ze mi ne, to bych si všim', proč já se trochu nesoustředím?\\
\hspace*{20pt}Teď jsem to měl, a je to pryč.\\
\textbf{B:} Tak mi to, švagře, trochu vylič.}
}

\bridge{
\textbf{A:} \chord{A}Nahoře to má, jako, takový \chord{Dm}čudlíky.\\
\textbf{B:} \chord{H\7}Ach, švagře, ty jsi pako. \chord{E\7}Jak je to veliký?\\
\textbf{A:} \chord{G}Podívej,
}

\vers{3}{
\textbf{A:} \rep{Bejt to velký, a ne malý, to bysme to tak nehledali.\\
\textbf{B:} To jsou trable, takhle k stáru, třebas' to zakop' pod almaru.}
}

\vers{4}{
\textbf{A:} Tam jsem byl, pod tou skříní, tam jsou jsou jen pavouci.\\
\textbf{B:} Řeknu ti, švagře, nyní tu pravdu pravdoucí.\\
\hspace*{20pt}Jak dobře víš, s tebou je kříž, ty tvoje věci maj' někde skrýš.\\
\hspace*{20pt}Jelikož jsou ale němé, tak už je nikdy nenajdeme.
}

\vers{5}{
\rep{Takovej ten, s takovou tou, jak to jen říci jednou větou?\\
Taková ta, s takovým tím,\\
\textbf{A:} Já tomu, švagře, nerozumím.}
}

\cod{Kde to je, teď před chvílí jsem to tu ještě měl.}
