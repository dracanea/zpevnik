\vers{2}{
S tebou je v nic, co srdce libovalo,\\
rozkotaná je naděje s tebou,\\
mi ji na místě, kde mi štěstí přálo,\\
\rep{pykaje, slasti an ztraceny jsou.}
}

\vers{3}{
Nemožno bylo-li mně libo zdáti,\\
proč stihnul mne vzezření tvého proud?\\

\rep{pak nekalil by mír můj žádný soud.}
}

\vers{4}{
Na srdci mě nevyjde nikdy více,\\
vše srdce ztrápené ti odpouští.\\

\rep{do tiché schránky mě doprovodí.}
}
